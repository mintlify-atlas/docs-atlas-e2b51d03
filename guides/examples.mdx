This guide provides comprehensive examples for common VirtualizorJS operations. All examples use actual SDK methods and demonstrate best practices.

## Getting Started

### Client Initialization

Before using any of the examples below, you'll need to create a client:

```typescript
import { createVirtualizorClient } from 'virtualizorjs';

const client = createVirtualizorClient({
  host: 'virtualizor.example.com',
  apiKey: 'your-api-key',
  apiPass: 'your-api-pass',
  port: 4085,        // 4085 for HTTPS, 4084 for HTTP
  https: true,       // Use HTTPS (default)
  timeout: 30000,    // Request timeout in ms
});
```

<Accordion title="Create a New VPS" icon="plus">

Create a new VPS instance with custom specifications:

<CodeGroup>

```typescript Basic Creation
import { VirtualizorApiError } from 'virtualizorjs';

try {
  // Create VPS
  const result = await client.vps.create({
    hostname: 'my-new-server.example.com',
    rootpass: 'SecurePassword123!',
    osid: 123,              // OS template ID
    plid: 1,                // Plan ID
    user_email: 'user@example.com',
    ips: 1,                 // Number of IPv4 addresses
    space: 50,              // Disk space in GB
    ram: 2048,              // RAM in MB
    bandwidth: 500,         // Bandwidth in GB
    cpu: 2,                 // CPU cores
    virt: 'kvm',            // Virtualization type
  });

  console.log('VPS creation initiated!');
  console.log(`Task ID: ${result.taskid}`);

  // Wait for the task to complete
  const task = await client.tasks.wait(result.taskid!, {
    pollIntervalMs: 5000,   // Check every 5 seconds
    timeoutMs: 600000,      // Timeout after 10 minutes
  });

  console.log('VPS created successfully!');
  console.log(`Task status: ${task.status}`);

} catch (err) {
  if (err instanceof VirtualizorApiError) {
    console.error(`API Error [${err.code}]: ${err.message}`);
  } else {
    console.error('Unexpected error:', err);
  }
}
```

```typescript Advanced Options
// Create VPS with SSH key and recipe
const result = await client.vps.create({
  hostname: 'app-server.example.com',
  rootpass: 'SecurePassword123!',
  osid: 456,
  plid: 2,
  user_email: 'admin@example.com',
  
  // Resource allocation
  space: 100,
  ram: 4096,
  burst: 512,             // Burst RAM
  bandwidth: 1000,
  cpu: 4,
  cpu_percent: 100,       // CPU percentage
  
  // Network configuration
  ips: 2,                 // Multiple IPs
  ips_int: 1,             // Internal IPs
  
  // Advanced options
  virt: 'kvm',
  serid: 5,               // Specific server
  recipe: 10,             // Post-install recipe
  sshkey: 'ssh-rsa AAAAB3NzaC1yc2E...',
  nopassword: 0,          // Keep password authentication
});

await client.tasks.wait(result.taskid!);
console.log('Advanced VPS created successfully!');
```

<Tab title="Start VPS">

```typescript
const vpsId = '123';
```

<Accordion title="Error Handling Best Practices" icon="shield-check">

Comprehensive error handling for production use:

```typescript
import { VirtualizorApiError } from 'virtualizorjs';

async function createVpsWithRetry(params: any, maxRetries = 3) {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      console.log(`Attempt ${attempt} of ${maxRetries}`);
      
      const result = await client.vps.create(params);
      
      const task = await client.tasks.wait(result.taskid!, {
        pollIntervalMs: 5000,
        timeoutMs: 600000,
      });
      
      console.log('VPS created successfully!');
      return task;
      
    } catch (err) {
      if (err instanceof VirtualizorApiError) {
        console.error(`API Error [${err.code}]: ${err.message}`);
        
        // Don't retry on validation errors
        if (err.code >= 400 && err.code < 500) {
          throw err;
        }
      }
      
      if (attempt === maxRetries) {
        console.error('Max retries reached');
        throw err;
      }
      
      // Wait before retrying
      await new Promise(r => setTimeout(r, 5000 * attempt));
    }
  }
}

// Usage
try {
  await createVpsWithRetry({
    hostname: 'resilient-server.example.com',
    rootpass: 'SecurePassword123!',
    osid: 123,
    plid: 1,
  });
} catch (err) {
  console.error('Failed to create VPS:', err);
}
```

<Accordion title="Progress Monitoring" icon="spinner">

Monitor long-running operations with progress updates:

```typescript
async function createVpsWithProgress(params: any) {
  console.log('Creating VPS...');
  const result = await client.vps.create(params);
  
  const taskId = result.taskid!;
  console.log(`Task ID: ${taskId}`);
  
  const startTime = Date.now();
  const pollInterval = 5000;
  const timeout = 600000;
  
  while (Date.now() - startTime < timeout) {
    const task = await client.tasks.get(taskId);
    
    if (!task) {
      throw new Error('Task not found');
    }
    
    const elapsed = Math.round((Date.now() - startTime) / 1000);
    console.log(`[${elapsed}s] Task status: ${task.status}`);
    
    if (task.status === '1' || task.status === 'done') {
      console.log('VPS created successfully!');
      return task;
    }
    
    if (task.status === 'error' || task.status === '-1') {
      throw new Error('Task failed');
    }
    
    await new Promise(r => setTimeout(r, pollInterval));
  }
  
  throw new Error('Task timed out');
}

// Usage
await createVpsWithProgress({
  hostname: 'monitored-server.example.com',
  rootpass: 'SecurePassword123!',
  osid: 123,
  plid: 1,
});
```

<Accordion title="Resource Monitoring Loop" icon="chart-area">

Continuously monitor VPS resources:

```typescript
async function monitorVPS(vpsId: string, durationMs: number) {
  const startTime = Date.now();
  const interval = 60000; // Check every minute
  
  console.log(`Monitoring VPS ${vpsId} for ${durationMs / 1000}s`);
  
  while (Date.now() - startTime < durationMs) {
    try {
      const stats = await client.vps.stats(vpsId);
      const vps = await client.vps.get(vpsId);
      
      const timestamp = new Date().toISOString();
      console.log(`[${timestamp}] ${vps.hostname}`);
      console.log(`  Status: ${vps.suspended ? 'Suspended' : 'Active'}`);
      console.log(`  Bandwidth used: ${vps.used_bandwidth} GB`);
      
    } catch (err) {
      console.error('Error monitoring VPS:', err);
    }
    
    await new Promise(r => setTimeout(r, interval));
  }
  
  console.log('Monitoring complete');
}

// Monitor a VPS for 1 hour
await monitorVPS('123', 3600000);
```
