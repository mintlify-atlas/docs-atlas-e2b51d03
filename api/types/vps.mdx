## Usage Examples

### Importing Types

```typescript
import type {
  VPS,
  CreateVPSParams,
  RebuildVPSParams,
  ListVPSParams
} from 'virtualizor';
```

### Creating a VPS

```typescript
import type { CreateVPSParams } from 'virtualizor';

const params: CreateVPSParams = {
  hostname: 'web-server-01.example.com',
  rootpass: 'SecurePassword123!',
  osid: 123,
  plid: 1,
  ram: 2048,
  space: 50,
  bandwidth: 1000,
  cpu: 2,
  virt: 'kvm'
};

const result = await client.vps.create(params);
```

### Rebuilding a VPS

```typescript
import type { RebuildVPSParams } from 'virtualizor';

const rebuildParams: RebuildVPSParams = {
  osid: 456,
  newpass: 'NewPassword123!',
  conf: 1,
  format_primary: 1,
  recipe: 10
};

await client.vps.rebuild('12345', rebuildParams);
```

### Filtering VPS List

```typescript
import type { ListVPSParams } from 'virtualizor';

const filters: ListVPSParams = {
  vstype: 'kvm',
  vsstatus: 'online',
  plid: 1
};

const vpsList = await client.vps.list(filters);
```

### Type-Safe VPS Properties

```typescript
import type { VPS } from 'virtualizor';

function displayVPSInfo(vps: VPS): void {
  console.log(`VPS ID: ${vps.vpsid}`);
  console.log(`Hostname: ${vps.hostname}`);
  console.log(`OS: ${vps.os_name}`);
  console.log(`RAM: ${vps.ram} MB`);
  console.log(`Bandwidth: ${vps.bandwidth} GB`);
  
  if (vps.ips) {
    console.log('IP Addresses:');
    Object.entries(vps.ips).forEach(([key, ip]) => {
      console.log(`  ${key}: ${ip}`);
    });
  }
}
```
